                    election_frequency = 48

                target = STC

#south carolina proposes collaboration (nonagg -> puppet / nothing -> nonagg -> puppet)
country_event = {
    id = gra_scarolina.1
    title = gra_scarolina.1.t
    desc = gra_scarolina.1.desc
    fire_only_once = yes
        trigger = {
            country_exists = STC
            GRA = {
                OR = {
                    has_country_flag = GRA_nc_republicanchoice
                    has_country_flag = GRA_nc_democratchoice
                }
            }
        }

    
        option = {
            name = gra_scarolina.1.a
            custom_effect_tooltip = GRA_sc_nonagg_puppet
            trigger = { GRA = { has_non_aggression_pact_with = STC } }
            hidden_effect = {
                STC = {
                    set_politics = {
                        ruling_party = conservative
                        elections_allowed = yes
                        election_frequency = 48
                    }
                }
                GRA = {
                    set_autonomy = {
                        target = STC
                        autonomy_state = autonomy_supervised_state
                    }
                }
            }

        }

        option = {
            name = gra_scarolina.1.c
            trigger = {
                GRA = { 
                    NOT = { 
                    has_non_aggression_pact_with = STC
                    has_subject = STC
                    } 
                } 
            }
            GRA = {
                diplomatic_relation = {
                    country = STC
                    relation = non_aggression_pact
                    active = yes
                }
                country_event = gra_scarolina.2
            }
        }
}
